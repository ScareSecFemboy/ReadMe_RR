Flags:
    --ph               | String   | Set a pixel height for GIF Injection 
    --pw               | String   | Set a pixel width  for GIF Injection
    --jpgc     / -p    | String   | Set a location chunk for JPG injection 
    --input    / -i    | String   | Set a input filename
    --output   / -o    | String   | Set a output filename
    --suppress / -s    | Bool     | Suppress hex output for metadata miners
    --offset           | String   | Set a location offset to inject data into a PNG image with
    --payload          | String   | Set a payload to inject into images or files
    --type             | String   | Set a chunk to inject at in a PNG image
    --filepath         | String   | Set a filepath for file parsing, logging, dumping etc
    --rrh              | String   | Spawn a flag help menu
    --help / -h        | Bool     | Run this menu
    --reso             | String   | Set a screen rotation for output <Verticle|Landscape|>
    --hasl             | String   | Set a hashlist for brute forcing or hash matching modules
    --userl            | String   | Set a username list for brute forcing modules
    --wordl            | String   | Set a wordlist for brute forcing modules 
    --workers          | Integer  | Set a number of workers for brute forcing modules
    --packet           | String   | Specify the packet type to sniff for network sniffing -> ICMP, tcp, eth, ip, dhcmp, application, dot11 are the only formats supported
    --interface        | String   | Specify the interface to use for network modules
    --targetm          | String   | Set a target mac address for arp poisoning
    --targetip         | String   | Set a target IP address for arp poisoning
    --gatemac          | String   | Set a targets gateway mac for arp poisoning
    --CIDR / -z        | String   | Set a network range to scan hosts on
    --passlen          | String   | Set a password length for password generation
    --perm             | String   | Set premission number for premission stomping
